540 def send_apns_notification(user:, title:, body:, data: {}, badge: nil, sound: "default", critical: false, interruption_level: "active", category: nil)
541 return unless user.devices.ios.any?
542 return unless APNS_CLIENT.present?
543
544 # Build notification payload
545 aps_payload = {
546 alert: {
547 title: title,
548 body: body
549 },
550 sound: sound,
551 badge: badge || calculate_badge_count(user)
552 }
553
554 # Add critical alert capability for iOS
555 if critical
556 aps_payload[:sound] = {
557 critical: 1,
558 name: "critical.caf",
559 volume: 1.0
560 }
561 end
562
563 # Add interruption level for iOS 15+
564 if interruption_level != "active"
565 aps_payload[:'interruption-level'] = interruption_level
566 end
567
568 # Add category for notification actions
569 if category
570 aps_payload[:category] = category
571 end
572
573 # Build complete payload
574 payload = {
575 aps: aps_payload,
576 data: data.merge(
577 title: title,
578 body: body,
579 timestamp: Time.current.to_i
580 ).compact
581 }
582
583 # Send to all iOS devices
584 user.devices.ios.each do |device|
585 begin
586 # Use our custom APNs client
587 response = APNS_CLIENT.send_notification(
588 device.apns_token,
589 payload,
590 push_type: "alert",
591 priority: critical ? 10 : 5,
592 expiration: Time.now.to_i + 1.hour.to_i
593 )
594
595 Rails.logger.info "[NotificationService] APNs sent to user ##{user.id}: #{title}"
596 Rails.logger.debug "[NotificationService] APNs Response: #{response.inspect}"
597
598 # Check for errors
599 if response[:status] == 410 # Unregistered
600 Rails.logger.warn "[NotificationService] Unregistered APNs token for user ##{user.id}, removing device..."
601 device.destroy
602 elsif !response[:ok]
603 Rails.logger.error "[NotificationService] APNs error for user ##{user.id}: #{response[:status]} - #{response[:reason]}"
604 end
605
606 rescue => e
607 Rails.logger.error "[NotificationService] APNs error sending to user ##{user.id}: #{e.message}"
608 Rails.logger.error e.backtrace.join("\n")
609 end
610 end
611 end

613 def send_fcm_notification(user:, title:, body:, data: {}, priority: "normal", critical: false, sound: "default", action: nil)
614 return unless user.fcm_token.present?
615
616 # Build notification payload
617 notification_payload = {
618 title: title,
619 body: body,
620 sound: sound
621 }
622
623 # Add critical alert capability for iOS
624 if critical
625 notification_payload[:sound] = {
626 critical: 1,
627 name: "critical.caf",
628 volume: 1.0
629 }
630 end
631
632 # Build complete payload
633 payload = {
634 notification: notification_payload,
635 data: data.merge(
636 title: title,
637 body: body,
638 action: action,
639 timestamp: Time.current.to_i
640 ).compact,
641 priority: priority,
642 content_available: true,
643 mutable_content: true # Allows notification service extension
644 }
645
646 # Add iOS-specific options
647 payload[:apns] = {
648 payload: {
649 aps: {
650 sound: critical ? "critical.caf" : "default",
651 badge: calculate_badge_count(user)
652 }
653 },
654 headers: {
655 'apns-priority': priority == "high" ? "10" : "5",
656 'apns-push-type': "alert"
657 }
658 }
659
660 if critical
661 payload[:apns][:payload][:aps]["sound"] = {
662 critical: 1,
663 name: "critical.caf",
664 volume: 1.0
665 }
666 end
667
668 begin
669 response = FCM_CLIENT.send([ user.fcm_token ], payload)
670
671 Rails.logger.info "[NotificationService] FCM sent to user ##{user.id}: #{title}"
672 Rails.logger.debug "[NotificationService] FCM Response: #{response.inspect}"
673
674 # Check if token is invalid
675 if response[:status_code] == 200
676 results = response[:body]["results"]
677 if results.any? { |r| r["error"] == "InvalidRegistration" || r["error"] == "NotRegistered" }
678 Rails.logger.warn "[NotificationService] Invalid FCM token for user ##{user.id}, clearing..."
679 user.update_column(:fcm_token, nil)
680 end
681 end
682
683 response
684
685 rescue => e
686 Rails.logger.error "[NotificationService] FCM error sending to user ##{user.id}: #{e.message}"
687 Rails.logger.error e.backtrace.join("\n")
688 { error: e.message }
689 end
690 end

708 def calculate_badge_count(user)
709 # Return count of incomplete un-snoozable tasks
710 Task.joins(:list)
711 .where(lists: { user_id: user.id })
712 .where(completed_at: nil)
713 .where(can_be_snoozed: false)
714 .count
715 end
