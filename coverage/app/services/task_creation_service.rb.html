22 def normalize_parameters

31 def handle_ios_parameters
32 # iOS uses 'name' instead of 'title'
33 @attrs[:title] = @params[:name] if @params[:name].present?
34
35 # iOS uses 'description' instead of 'note'

40 end

42 def handle_date_parameters
43 @attrs[:due_at] = extract_due_at(@params)
44 end
45
46 def extract_due_at(h)
47 return parse_epoch(h[:dueDate]) if h.key?(:dueDate) # epoch seconds
48 return parse_time(h[:due_date]) if h.key?(:due_date) # ISO8601
49 return parse_time(h[:due_at]) if h.key?(:due_at) # ISO8601
50 nil
51 rescue ArgumentError
52 nil
53 end
54
55 def parse_epoch(v)
56 return nil if v.blank?
57 Time.at(v.to_i)
58 rescue ArgumentError
59 nil
60 end
61
62 def parse_time(v)
63 return nil if v.blank?
64 case v
65 when Time then v.in_time_zone
66 when String
67 begin
68 Time.iso8601(v).in_time_zone # preserves instant if "Z"/offset included
69 rescue ArgumentError
70 Time.zone.parse(v.to_s)
71 end
72 end
73 rescue StandardError
74 nil
75 end

87 def boolean(value)
88 case value
89 when true, false then value
90 when "true", "1", 1 then true
91 when "false", "0", 0, nil then false

105 def create_task
106 @task = @list.tasks.build(@attrs)
107 @task.creator = @user
108
109 # Set default due_at if not provided and required
110 if @task.due_at.blank? && @task.strict_mode
111 @task.due_at = 1.hour.from_now

115 end

117 def create_subtasks(subtasks)
118 subtasks.each do |subtask_title|
119 # Truncate title if too long
120 title = subtask_title.to_s
121 title = title[0, 255] if title.length > 255

123 @task.subtasks.create!(
