95 # Safe time period parsing
96 def self.parse_time_period(period_string)
97 case period_string.downcase.strip
98 when /^(\d+)\s*minutes?$/
99 $1.to_i.minutes
100 when /^(\d+)\s*hours?$/
101 $1.to_i.hours
102 when /^(\d+)\s*days?$/
103 $1.to_i.days
104 when /^(\d+)\s*weeks?$/
105 $1.to_i.weeks
106 when /^(\d+)\s*months?$/
107 $1.to_i.months
108 when /^(\d+)\s*years?$/
109 $1.to_i.years
110 else
111 # Fallback to default if parsing fails
112 1.week
113 end
114 end
