451 def calculate_daily_recurrence
452 base_time = recurrence_time || (due_at || Time.current)
453 day_anchor = (due_at || Time.current).beginning_of_day
454 next_time = day_anchor + base_time.seconds_since_midnight
455 next_time += 1.day if next_time <= Time.current
456 next_time
457 end

459 def calculate_weekly_recurrence
460 return nil unless recurrence_days.present?
461 base_time = recurrence_time || (due_at || Time.current)
462 days = recurrence_days.map(&:to_i).sort
463 now = Time.current
464 days.each do |wday|
465 candidate = now.beginning_of_week + wday.days
466 candidate = candidate.change(hour: base_time.hour, min: base_time.min, sec: base_time.sec)
467 return candidate if candidate > now
468 end
469 # next week
470 first = days.first
471 (now.beginning_of_week + 1.week + first.days).change(
472 hour: base_time.hour, min: base_time.min, sec: base_time.sec
473 )
474 end
475
476 def calculate_monthly_recurrence
477 base_time = recurrence_time || (due_at || Time.current)
478 now = Time.current
479 candidate = now.change(day: (due_at || now).day, hour: base_time.hour, min: base_time.min, sec: base_time.sec) rescue nil
480 candidate = (now.beginning_of_month + base_time.seconds_since_midnight) unless candidate
481 candidate = candidate.next_month if candidate <= now
482 candidate
483 end
484
485 def calculate_yearly_recurrence
486 base_time = recurrence_time || (due_at || Time.current)
487 now = Time.current
488 candidate = now.change(month: (due_at || now).month, day: (due_at || now).day,
489 hour: base_time.hour, min: base_time.min, sec: base_time.sec) rescue nil
490 candidate ||= now.beginning_of_year + base_time.seconds_since_midnight
491 candidate = candidate.next_year if candidate <= now
492 candidate
493 end

495 # == Utilities
496 def calculate_distance(lat1, lon1, lat2, lon2)
497 return 0 if lat1 == lat2 && lon1 == lon2
498 to_rad = Math::PI / 180.0
499 lat1 *= to_rad; lon1 *= to_rad
500 lat2 *= to_rad; lon2 *= to_rad
501 dlat = lat2 - lat1
502 dlon = lon2 - lon1
503 a = Math.sin(dlat / 2)**2 + Math.cos(lat1) * Math.cos(lat2) * Math.sin(dlon / 2)**2
504 c = 2 * Math.asin(Math.sqrt(a))
505 6_371_000 * c
506 end
