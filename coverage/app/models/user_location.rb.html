56 # distance to coordinates (meters) – Haversine
57 def distance_to_coordinates(lat, lng)
58 return 0 if lat == latitude && lng == longitude
59
60 rad = Math::PI / 180.0
61 r_km = 6371.0
62
63 dlat = (lat - latitude) * rad
64 dlng = (lng - longitude) * rad
65 lat1 = latitude * rad
66 lat2 = lat * rad
67
68 a = Math.sin(dlat / 2)**2 +
69 Math.cos(lat1) * Math.cos(lat2) * Math.sin(dlng / 2)**2
70 c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a))
71 (r_km * c * 1000).round
72 end
