185 def ensure_unique_summary_date
186 return if summary_date == nil # Don't auto-set if explicitly set to nil
187 return if summary_date.blank? # Don't auto-set if no date provided
188
189 # Only auto-adjust if there's a conflict and we're not in a validation context
190 if self.class.exists?(coaching_relationship_id: coaching_relationship_id, summary_date: summary_date) && !@_validating_uniqueness
191 date = summary_date
192 # pick the most recent available date going backwards
193 while self.class.exists?(coaching_relationship_id: coaching_relationship_id, summary_date: date)
194 date -= 1.day
195 end
196 self.summary_date = date
197 end
