295 # Calculate distance between two points using Haversine formula
296 def calculate_distance(lat1, lon1, lat2, lon2)
297 # Earth's radius in meters
298 earth_radius = 6_371_000
299
300 # Convert degrees to radians
301 lat1_rad = lat1 * Math::PI / 180
302 lat2_rad = lat2 * Math::PI / 180
303 delta_lat = (lat2 - lat1) * Math::PI / 180
304 delta_lon = (lon2 - lon1) * Math::PI / 180
305
306 # Haversine formula
307 a = Math.sin(delta_lat / 2) ** 2 +
308 Math.cos(lat1_rad) * Math.cos(lat2_rad) *
309 Math.sin(delta_lon / 2) ** 2
310 c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a))
311
312 earth_radius * c
313 end
