83 def derive_priority
84 return 3 if task.strict_mode && overdue? # Urgent
85 return 2 if task.strict_mode # High
86 return 1 if task.due_at && task.due_at < 24.hours.from_now # Medium
87 0 # Low
88 end
89
90 def overdue?
91 # Task is overdue if it has a due date in the past and is NOT completed
92 task.due_at.present? &&
93 task.due_at < Time.current &&
94 (task.status.nil? || task.status == "pending" || task.status == "in_progress")
95 end
96
97 def minutes_overdue
98 return 0 unless overdue?
99 ((Time.current - task.due_at) / 60).to_i
100 end

102 def requires_explanation?
