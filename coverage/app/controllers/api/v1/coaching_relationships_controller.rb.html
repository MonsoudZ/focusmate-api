26 # POST /api/v1/coaching_relationships
27 def create
28 coach_email = normalize_email(params[:coach_email])
29 client_email = normalize_email(params[:client_email])
30 invited_by = params[:invited_by].presence
31
32 if coach_email.blank? && client_email.blank?
33 return render json: { error: { message: "Must provide coach_email or client_email" } }, status: :unprocessable_entity
34 end
35
36 if invited_by == "coach" || (invited_by.blank? && client_email.present?)
37 # coach invites client
38 coach = current_user
39 client = find_user_ci(client_email || coach_email)
40 return not_found!("Client not found with that email") unless client
41 return render json: { error: { message: "You cannot invite yourself" } }, status: :unprocessable_entity if client.id == coach.id
42 invited_by = "coach"
43 else
44 # client invites coach
45 client = current_user
46 coach = find_user_ci(coach_email || client_email)
47 return not_found!("Coach not found with that email") unless coach
48 return render json: { error: { message: "You cannot invite yourself" } }, status: :unprocessable_entity if coach.id == client.id
49 invited_by = "client"
50 end
51
52 if CoachingRelationship.exists?(coach_id: coach.id, client_id: client.id)
53 return render json: { error: { message: "Relationship already exists" } }, status: :unprocessable_entity
54 end
55
56 rel = CoachingRelationship.new(
57 coach: coach,
58 client: client,
59 invited_by: invited_by,
60 status: "pending"
61 )
62
63 if rel.save
64 safe_notify(:coaching_invitation_sent, rel)
65 render json: CoachingRelationshipSerializer.new(rel, current_user: current_user).as_json, status: :created
66 else
67 validation_error!(rel.errors)
68 end
69 end

93 # PATCH /api/v1/coaching_relationships/:id/update_preferences
94 def update_preferences
95 return not_found! unless @relationship && participant?(@relationship, current_user)
96 return forbidden!("Only coaches can update preferences") unless current_user.id == @relationship.coach_id
97
98 src = params[:coaching_relationship].is_a?(ActionController::Parameters) ? params[:coaching_relationship] : params
99
100 changes = {}
101 changes[:notify_on_completion] = to_bool(src[:notify_on_completion]) if src.key?(:notify_on_completion)
102 changes[:notify_on_missed_deadline] = to_bool(src[:notify_on_missed_deadline]) if src.key?(:notify_on_missed_deadline)
103 changes[:send_daily_summary] = to_bool(src[:send_daily_summary]) if src.key?(:send_daily_summary)
104
105 if src.key?(:daily_summary_time)
106 if src[:daily_summary_time].present?
107 hhmm = parse_hhmm(src[:daily_summary_time])
108 return render json: { error: { message: "Validation failed", details: { daily_summary_time: [ "must be in HH:MM format" ] } } }, status: :unprocessable_entity unless hhmm
109 changes[:daily_summary_time] = hhmm
110 else
111 # allow clearing with empty string / nil
112 changes[:daily_summary_time] = nil
113 end
114 end
115
116 # Persist either in serialized preferences hash or direct attributes
117 if @relationship.respond_to?(:preferences) && @relationship.preferences.is_a?(Hash)
118 @relationship.preferences = (@relationship.preferences || {}).merge(changes.stringify_keys)
119 else
120 changes.each do |k, v|
121 setter = "#{k}="
122 if @relationship.respond_to?(setter)
123 if k == :daily_summary_time && v.present?
124 # Convert string time to Time object for database storage
125 time_parts = v.split(":")
126 time_obj = Time.zone.parse("#{time_parts[0]}:#{time_parts[1]}")
127 @relationship.public_send(setter, time_obj)
128 else
129 @relationship.public_send(setter, v)
130 end
131 end
132 end
133 end
134
135 @relationship.save!
136
137 render json: {
138 id: @relationship.id,
139 notify_on_completion: current_pref(:notify_on_completion),
140 notify_on_missed_deadline: current_pref(:notify_on_missed_deadline),
141 send_daily_summary: current_pref(:send_daily_summary),
142 daily_summary_time: format_hhmm(current_pref(:daily_summary_time))
143 }, status: :ok
144 end

193 def find_user_ci(email)
194 e = email.to_s.strip
195 return nil if e.blank?
196 User.where("LOWER(email) = ?", e.downcase).first
197 end

199 def to_bool(val)
200 return val if val == true || val == false
201 s = val.to_s.strip.downcase
202 return true if %w[true 1 yes y t on].include?(s)
203 return false if %w[false 0 no n f off].include?(s)
204 nil
205 end

207 # returns "HH:MM" or nil for invalid
208 def parse_hhmm(val)
209 s = val.to_s.strip
210 return nil unless s =~ /\A([01]?\d|2[0-3]):[0-5]\d\z/
211 hh, mm = s.split(":")
212 format("%02d:%02d", hh.to_i, mm.to_i)
213 end
214
215 # normalize stored value into "HH:MM" or nil
216 def format_hhmm(val)
217 return nil if val.blank?
218 return val if val.is_a?(String) && val.match?(/\A([01]?\d|2[0-3]):[0-5]\d\z/)
219 if val.respond_to?(:strftime)
220 return val.strftime("%H:%M")
221 end
222 s = val.to_s
223 return s if s.match?(/\A([01]?\d|2[0-3]):[0-5]\d\z/)
224 nil
225 end
