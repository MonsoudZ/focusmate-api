96 def calculate_completion_rate(user, since)
97 tasks = Task.joins(:list)
98 .where(lists: { user_id: user.id })
99 .where("tasks.created_at >= ?", since)
100
101 return 0 if tasks.none?
102
103 (tasks.where(status: :done).count.to_f / tasks.count * 100).round(1)
104 end

106 def calculate_average_completion_time(user)
107 tasks = Task.joins(:list)
108 .where(lists: { user_id: user.id })
109 .where.not(completed_at: nil)
110 .where.not(due_at: nil)
111
112 return 0 if tasks.none?
113
114 total_diff = tasks.sum { |t| (t.completed_at - t.due_at).abs }
115 (total_diff / tasks.count / 3600).round(1) # Convert to hours
116 end

178 def calculate_average_client_completion_rate
179 clients = current_user.clients
180 return 0 if clients.empty?
181
182 # Optimize with single query instead of N+1
183 client_ids = clients.pluck(:id)
184 task_stats = Task.joins(:list)
185 .where(lists: { user_id: client_ids })
186 .group("lists.user_id")
187 .select("lists.user_id, COUNT(*) as total_tasks, SUM(CASE WHEN tasks.status = 1 THEN 1 ELSE 0 END) as completed_tasks")
188
189 stats_by_client = task_stats.index_by(&:user_id)
190
191 total_completion_rate = clients.sum do |client|
192 stats = stats_by_client[client.id]
193 next 0 unless stats && stats.total_tasks.to_i > 0
194
195 (stats.completed_tasks.to_f / stats.total_tasks.to_i * 100).round(1)
196 end
197
198 (total_completion_rate / clients.count).round(1)
199 end

201 def client_performance_summary
202 # Optimize with single query instead of N+1
203 clients = current_user.clients
204 client_ids = clients.pluck(:id)
205
206 # Single query to get all task statistics
207 task_stats = Task.joins(:list)
208 .where(lists: { user_id: client_ids })
209 .group("lists.user_id")
210 .select('lists.user_id,
211 COUNT(*) as total_tasks,
212 SUM(CASE WHEN tasks.status = 1 THEN 1 ELSE 0 END) as completed_tasks,
213 SUM(CASE WHEN tasks.status = 0 AND tasks.due_at < NOW() THEN 1 ELSE 0 END) as overdue_tasks')
214
215 stats_by_client = task_stats.index_by(&:user_id)
216
217 clients.map do |client|
218 stats = stats_by_client[client.id]
219
220 {
221 client_id: client.id,
222 client_name: client.name,
223 total_tasks: stats&.total_tasks || 0,
224 completed_tasks: stats&.completed_tasks || 0,
225 overdue_tasks: stats&.overdue_tasks || 0,
226 completion_rate: (stats && stats.total_tasks.to_i > 0) ? (stats.completed_tasks.to_f / stats.total_tasks.to_i * 100).round(1) : 0
227 }
228 end
