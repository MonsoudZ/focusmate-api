51 # POST /api/v1/lists/:list_id/memberships
52 def create
53 # Find user by email or ID
54 target_user = find_user_by_email_or_id(membership_params[:user_identifier])
55
56 unless target_user
57 return render json: { error: "User not found" }, status: :not_found
58 end
59
60 # Check if user is already a member
61 if @list.members.include?(target_user)
62 return render json: { error: "User is already a member of this list" }, status: :unprocessable_entity
63 end
64
65 # Check if user is trying to invite themselves
66 if target_user == current_user
67 return render json: { error: "Cannot invite yourself" }, status: :unprocessable_entity
68 end
69
70 @membership = @list.memberships.build(
71 user: target_user,
72 role: membership_params[:role] || "viewer"
73 )
74
75 authorize @membership
76
77 if @membership.save
78 render json: {
79 membership: {
80 id: @membership.id,
81 user: {
82 id: @membership.user.id,
83 email: @membership.user.email
84 },
85 role: @membership.role,
86 created_at: @membership.created_at,
87 updated_at: @membership.updated_at
88 }
89 }, status: :created
90 else
91 render json: { errors: @membership.errors.full_messages }, status: :unprocessable_entity
92 end
93 end

135 def membership_params
136 # Only permit user_identifier, handle role separately for security
137 permitted = params.require(:membership).permit(:user_identifier)
138
139 # Handle role separately with explicit validation
140 if params[:membership][:role].present?
141 role = params[:membership][:role].to_s.downcase
142 if %w[editor viewer].include?(role)
143 permitted[:role] = role
144 else
145 permitted[:role] = "viewer" # Default to viewer for invalid roles
146 end
147 else
148 permitted[:role] = "viewer" # Default role
149 end

151 permitted

154 def find_user_by_email_or_id(identifier)
155 # Try to find by ID first (if it's a number)
156 if identifier.match?(/^\d+$/)
157 User.find_by(id: identifier)
158 else
159 # Otherwise, try to find by email
160 User.find_by(email: identifier)
