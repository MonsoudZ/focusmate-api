7 # iOS can call this as PUT /devices/token or PATCH /users/device_token
8 def update_device_token
9 # Handle both parameter formats
10 token = params[:device_token] || params[:pushToken] || params[:push_token]
11
12 # Allow nil tokens for logout, but reject empty/whitespace tokens
13 if token.blank? && !params.key?(:device_token) && !params.key?(:pushToken) && !params.key?(:push_token)
14 render json: { error: { message: "Device token is required" } }, status: :bad_request
15 return
16 end
17
18 # Reject empty or whitespace-only tokens (but allow nil for logout)
19 if token && token.strip.blank?
20 render json: { error: { message: "Device token is required" } }, status: :bad_request
21 return
22 end
23
24 if current_user.update(device_token: token)
25 token_preview = token.present? ? "#{token[0..20]}..." : "nil (logout)"
26 Rails.logger.info "[DeviceToken] Updated for user ##{current_user.id}: #{token_preview}"
27 render json: {
28 message: "Device token updated successfully",
29 user_id: current_user.id
30 }, status: :ok
31 else
32 render json: {
33 error: "Failed to update device token",
34 details: current_user.errors.full_messages
35 }, status: :unprocessable_entity
36 end
37 end

39 def update_location
40 latitude = params[:latitude]
41 longitude = params[:longitude]
42
43 if latitude.blank? || longitude.blank?
44 render json: { error: { message: "Latitude and longitude are required" } }, status: :bad_request
45 return
46 end
47
48 Rails.logger.info "Updating location for user #{current_user.id}: lat=#{latitude}, lng=#{longitude}"
49
50 if current_user.update(
51 latitude: latitude.to_f,
52 longitude: longitude.to_f,
53 location_updated_at: Time.current
54 )
55 # Create UserLocation record for history tracking
56 current_user.user_locations.create!(
57 latitude: latitude.to_f,
58 longitude: longitude.to_f,
59 recorded_at: Time.current
60 )
61
62 current_user.reload
63 Rails.logger.info "Location updated successfully: lat=#{current_user.latitude}, lng=#{current_user.longitude}"
64 render json: {
65 message: "Location updated successfully",
66 latitude: current_user.latitude,
67 longitude: current_user.longitude
68 }, status: :ok
69 else
70 Rails.logger.error "Failed to update location: #{current_user.errors.full_messages}"
