86 # POST /api/v1/lists/:id/share
87 # params: user_id or email (required), can_edit (optional)
88 def share
89 target_id = params[:user_id]
90 target_email = params[:email]
91
92 # Check if both user_id and email are missing
93 if target_id.blank? && target_email.blank?
94 return render json: { errors: { email: [ "is required" ] } }, status: :unprocessable_entity
95 end
96
97 # If email is provided, find user by email
98 if target_email.present?
99 user = User.find_by(email: target_email)
100 return render json: { errors: { email: [ "User not found" ] } }, status: :unprocessable_entity unless user
101 target_id = user.id
102 end
103
104 # Get user's email for ListShare model
105 user = User.find(target_id)
106 share = ListShare.find_or_initialize_by(list_id: @list.id, email: user.email)
107 share.user_id = target_id
108 share.status = :accepted
109 share.can_view = ActiveModel::Type::Boolean.new.cast(params[:can_view])
110 share.can_edit = ActiveModel::Type::Boolean.new.cast(params[:can_edit])
111 share.can_add_items = ActiveModel::Type::Boolean.new.cast(params[:can_add_items])
112 share.can_delete_items = ActiveModel::Type::Boolean.new.cast(params[:can_delete_items])
113
114 if share.save
115 render json: {
116 id: share.id,
117 user_id: share.user_id,
118 email: share.email,
119 can_view: share.can_view,
120 can_edit: share.can_edit,
121 can_add_items: share.can_add_items,
122 can_delete_items: share.can_delete_items,
123 status: share.status,
124 created_at: share.created_at,
125 updated_at: share.updated_at
126 }, status: :created
127 else
128 validation_error!(share)
129 end
130 end
