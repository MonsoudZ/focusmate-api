3module ErrorLoggingHelper
4 extend ActiveSupport::Concern
5
6 included do
7 rescue_from StandardError, with: :handle_unexpected_error
8 rescue_from ActiveRecord::RecordNotFound, with: :handle_record_not_found
9 rescue_from ActiveRecord::RecordInvalid, with: :handle_record_invalid
10 rescue_from Pundit::NotAuthorizedError, with: :handle_authorization_error
11 end
12
13 private
14
15 def handle_unexpected_error(exception)
16 log_error(exception, severity: :error)
17 render_server_error("An unexpected error occurred")
18 end
19
20 def handle_record_not_found(exception)
21 log_error(exception, severity: :warn)
22 render_not_found("Resource")
23 end
24
25 def handle_record_invalid(exception)
26 log_error(exception, severity: :info)
27 render_validation_errors(exception.record.errors)
28 end
29
30 def handle_authorization_error(exception)
31 log_error(exception, severity: :warn)
32 render_forbidden("You do not have permission to perform this action")
33 end
34
35 def log_error(exception, severity: :error)
36 error_data = {
37 exception: exception.class.name,
38 message: exception.message,
39 backtrace: Rails.env.development? ? exception.backtrace : nil,
40 user_id: current_user&.id,
41 request_id: request.request_id,
42 params: sanitized_params,
43 timestamp: Time.current.iso8601
44 }
45
46 case severity
47 when :error
48 Rails.logger.error "API Error: #{error_data.to_json}"
49 when :warn
50 Rails.logger.warn "API Warning: #{error_data.to_json}"
51 when :info
52 Rails.logger.info "API Info: #{error_data.to_json}"
53 end
54
55 # In production, you might want to send to external error tracking service
56 # Sentry.capture_exception(exception) if defined?(Sentry)
57 end
58
59 def sanitized_params
60 # Remove sensitive parameters from logging
61 sensitive_keys = %w[password password_confirmation token authorization]
62 params.except(*sensitive_keys).to_unsafe_h
63 end
