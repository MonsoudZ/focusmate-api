6 # Standard error response format
7 def render_error(message, status = :bad_request, details = nil)
8 error_response = {
9 error: {
10 message: message,
11 status: Rack::Utils.status_code(status),
12 timestamp: Time.current.iso8601
13 }
14 }
15
16 error_response[:error][:details] = details if details.present?

49 error_response = {

52 status: 429,
