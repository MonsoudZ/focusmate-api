23 def generate_instances_for_template(template)
24 # Calculate when the next instance should be due
25 next_due = template.calculate_next_due_date
26
27 return unless next_due
28
29 # Check if we've reached the end date
30 if template.recurrence_end_date.present? && next_due > template.recurrence_end_date
31 Rails.logger.info "[RecurringTaskGeneratorWorker] Template ##{template.id} has reached end date"
32 return
33 end
34
35 # Generate instances for the next 7 days
36 instances_to_generate = []
37 current_date = next_due
38
39 7.times do
40 break if template.recurrence_end_date.present? && current_date > template.recurrence_end_date
41
42 # Check if instance already exists for this date
43 existing = template.recurring_instances.find_by(
44 "DATE(due_at) = ?", current_date.to_date
45 )
46
47 if existing.nil?
48 instances_to_generate << current_date
49 end
50
51 # Calculate next date based on pattern
52 current_date = advance_date(current_date, template)
53 end
54
55 # Create the instances
56 instances_to_generate.each do |due_date|
57 create_instance(template, due_date)
58 end
59
60 if instances_to_generate.any?
61 Rails.logger.info "[RecurringTaskGeneratorWorker] Generated #{instances_to_generate.count} instances for template ##{template.id}"
62 end
63

105 def advance_date(date, template)
106 case template.recurrence_pattern
107 when "daily"
108 date + template.recurrence_interval.days
109 when "weekly"
110 next_date = date + template.recurrence_interval.weeks
111
112 # If specific days are set, advance to next matching day
113 if template.recurrence_days.present?
114 while !template.recurrence_days.include?(next_date.wday)
115 next_date += 1.day
116 end
117 end
118
119 next_date
120 when "monthly"
121 date + template.recurrence_interval.months
122 when "custom"
123 date + template.recurrence_interval.days
124 else
125 date + 1.day
126 end
