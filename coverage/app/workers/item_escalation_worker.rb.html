25 def process_task_escalation(task)
26 escalation = task.escalation || task.create_escalation!
27
28 # Calculate time since last notification
29 time_since_last = if escalation.last_notification_at
30 (Time.current - escalation.last_notification_at) / 60.0
31 else
32 999 # Force notification if never sent
33 end
34
35 # Check if it's time to send another notification
36 if time_since_last >= task.notification_interval_minutes
37 Rails.logger.info "[ItemEscalationWorker] Escalating task ##{task.id}: '#{task.title}'"
38
39 # Increment notification count and update escalation level
40 escalation.increment!(:notification_count)
41 escalation.update!(last_notification_at: Time.current)
42
43 # Mark as overdue if not already
44 if escalation.became_overdue_at.nil?
45 escalation.update!(became_overdue_at: Time.current)
46 end
47
48 # Check and update escalation level
49 check_escalation_level(task, escalation)
50
51 # Send notification
52 NotificationService.send_reminder(task, escalation.escalation_level)
53
54 # Check if we should notify coaches (at critical level)
55 if escalation.escalation_level == "critical" && !escalation.coaches_notified?
56 escalation.update!(coaches_notified: true, coaches_notified_at: Time.current)
57 NotificationService.alert_coaches_of_overdue(task)
58 end
59
60 # Check if app should be blocked
61 if task.should_block_app? && !escalation.blocking_app?
62 Rails.logger.warn "[ItemEscalationWorker] Blocking app for task ##{task.id}"
63 start_blocking_app(task, escalation)
64 end
65 end
66
67 rescue => e
68 Rails.logger.error "[ItemEscalationWorker] Error processing task ##{task.id}: #{e.message}"
69 Rails.logger.error e.backtrace.join("\n")
70 end

99 elsif minutes_overdue > 120

102 "normal"
103 end
104 end

106 if new_level != escalation.escalation_level
